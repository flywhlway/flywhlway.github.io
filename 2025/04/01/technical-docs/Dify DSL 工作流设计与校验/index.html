#!/usr/bin/env python3
## -*- coding: utf-8 -*-

"""
Dify DSL Wizard (2-step)
Step1: build context_pack.yaml + prompt_template.md + export all DSLs + pick mother DSL
Step2: generate DSL from requirements.md via LLM -> local validate -> import to Dify (precheck) -> export normalized DSL as final deliverable
"""

import argparse
import base64
import datetime as dt
import json
import os
import re
import sys
from dataclasses import dataclass
from pathlib import Path
from typing import Any, Dict, List, Optional, Tuple

import requests
import yaml

## -----------------------------
## Utils
## -----------------------------
TIMEOUT = 30

def die(msg: str, code: int = 1) -> None:
    print(f"[ERROR] {msg}", file=sys.stderr)
    sys.exit(code)

def now_ts() -> str:
    return dt.datetime.now().strftime("%Y%m%d_%H%M%S")

def safe_filename(name: str) -> str:
    name = re.sub(r"[\\/:*?\"<>|]+", "_", name).strip()
    return name[:120] or "app"

def extract_data(payload: Any) -> Any:
    # Dify 常见：{"data": ...} 或直接返回对象/字符串
    if isinstance(payload, dict) and "data" in payload:
        return payload["data"]
    return payload

def read_yaml_file(path: Path) -> Dict[str, Any]:
    try:
        return yaml.safe_load(path.read_text(encoding="utf-8"))
    except Exception as e:
        die(f"Failed to read YAML config: {path} err={e}")

def write_text(path: Path, content: str) -> None:
    path.parent.mkdir(parents=True, exist_ok=True)
    path.write_text(content, encoding="utf-8")

def write_yaml(path: Path, obj: Any) -> None:
    path.parent.mkdir(parents=True, exist_ok=True)
    path.write_text(yaml.safe_dump(obj, sort_keys=False, allow_unicode=True), encoding="utf-8")

## Config
## -----------------------------
@dataclass
class DifyConfig:
    origin: str
    console_api: str
    email: str
    password: str

@dataclass
class LLMConfig:
    base_url: str
    api_key: str
    model: str

@dataclass
class AppPick:
    app_id: str
    app_name: str

def load_config(config_path: Path) -> Tuple[DifyConfig, LLMConfig]:
    cfg = read_yaml_file(config_path) or {}
    dify = cfg.get("dify") or {}
    llm = cfg.get("llm") or {}

    origin = str(dify.get("origin", "")).rstrip("/")
    if not origin:
        die("config.yaml missing dify.origin")

    console_api = str(dify.get("console_api", "")).rstrip("/")
    if not console_api:
        console_api = f"{origin}/console/api"

    email = str(dify.get("email", ""))
    password = str(dify.get("password", ""))
    if not email or not password:
        die("config.yaml missing dify.email or dify.password")

    base_url = str(llm.get("base_url", "")).rstrip("/")
    api_key = str(llm.get("api_key", ""))
    model = str(llm.get("model", ""))
    temperature = float(llm.get("temperature", 0.2))

    if not base_url or not api_key or not model:
        die("config.yaml missing llm.base_url or llm.api_key or llm.model")

    return (
        DifyConfig(origin=origin, console_api=console_api, email=email, password=password),
        LLMConfig(base_url=base_url, api_key=api_key, model=model, temperature=temperature),
    )

## -----------------------------
## Dify Console API Client
## -----------------------------
class DifyConsole:
    def __init__(self, cfg: DifyConfig):
        self.cfg = cfg
        self.session = requests.Session()
        self.token: Optional[str] = None

    def _headers(self) -> Dict[str, str]:
        h = {"Content-Type": "application/json"}
        if self.token:
            h["Authorization"] = f"Bearer {self.token}"
        return h

    def login(self) -> str:
        pwd_b64 = base64.b64encode(self.cfg.password.encode("utf-8")).decode("utf-8")
        url = f"{self.cfg.console_api}/login"
        r = self.session.post(url, headers={"Content-Type": "application/json"}, json={"email": self.cfg.email, "password": pwd_b64}, timeout=TIMEOUT)
        r.raise_for_status()
        data = extract_data(r.json())
        token = data.get("access_token") if isinstance(data, dict) else None
        if not token:
            die(f"Login failed, response={r.text}")
        self.token = token
        return token

    def get(self, path: str, params: Dict[str, Any] | None = None) -> Any:
        url = f"{self.cfg.console_api}{path}"
        r = self.session.get(url, headers=self._headers(), params=params, timeout=TIMEOUT)
        r.raise_for_status()
        return r.json()

    def post(self, path: str, payload: Dict[str, Any]) -> Tuple[int, Any]:
        url = f"{self.cfg.console_api}{path}"
        r = self.session.post(url, headers=self._headers(), json=payload, timeout=TIMEOUT)
        # import 可能返回 400/202/200，我们要保留 status_code
        try:
            body = r.json()
        except Exception:
            body = {"raw": r.text}
        return r.status_code, body

    def list_apps(self) -> List[Dict[str, Any]]:
        apps: List[Dict[str, Any]] = []
        page, limit = 1, 30
        while True:
            payload = self.get("/apps", params={"page": page, "limit": limit})
            data = extract_data(payload)
            batch = data.get("data", []) if isinstance(data, dict) else []
            total = data.get("total", 0) if isinstance(data, dict) else 0
            apps.extend(batch)
            if len(apps) >= total:
                break
            page += 1
        return apps

    def export_app_dsl(self, app_id: str, include_secret: bool = False) -> str:
        payload = self.get(f"/apps/{app_id}/export", params={"include_secret": "true" if include_secret else "false"})
        yml_text = extract_data(payload)
        if not isinstance(yml_text, str) or not yml_text.strip():
            die(f"Export DSL failed: app_id={app_id}, resp={payload}")
        return yml_text

    def get_model_providers(self) -> Any:
        return extract_data(self.get("/workspaces/current/model-providers"))

    def get_models_by_type(self, model_type: str) -> Any:
        return extract_data(self.get(f"/workspaces/current/models/model-types/{model_type}"))

    def get_plugins(self) -> Any:
        # 插件系统未启用可能会 404
        try:
            return extract_data(self.get("/workspaces/current/plugin/list"))
        except Exception as e:
            return {"error": f"plugins endpoint unavailable: {e}"}

    def get_version_probe(self) -> Any:
        # /version 可能接受 current_version 参数（不同部署略有差异）
        try:
            return extract_data(self.get("/version", params={"current_version": "0.0.0"}))
        except Exception as e:
            return {"error": str(e)}

### 9. 方案详述（来源：Dify DSL生成与优化）

## 自动化脚本（Python）：一键生成 `context_pack.yaml` + `prompt_template.md` + `dsl_exports/`

> 你可以直接把下面脚本丢到服务器运行（不依赖 Dify 内部环境，只要能访问 Dify 的 console 域名即可）。

```python
#!/usr/bin/env python3
## -*- coding: utf-8 -*-
"""
Generate:
- out/context_pack.yaml
- out/prompt_template.md
- out/dsl_exports/*.yml
- out/dsl_coverage_report.md
"""
import os, base64, pathlib, re
from typing import Any, Dict, List, Tuple

import requests
import yaml

TIMEOUT = 30

def http_json(method: str, url: str, token: str | None = None, **kwargs) -> Dict[str, Any]:
    headers = kwargs.pop("headers", {})
    if token:
        headers["Authorization"] = f"Bearer {token}"
    r = requests.request(method, url, headers=headers, timeout=TIMEOUT, **kwargs)
    r.raise_for_status()
    return r.json()

def extract_data(payload: Dict[str, Any]) -> Any:
    # Dify console API常见返回：{"data": ...} 或直接返回对象
    return payload.get("data", payload)

def login(console_api: str, email: str, password_plain: str) -> str:
    pwd_b64 = base64.b64encode(password_plain.encode("utf-8")).decode("utf-8")
    payload = http_json(
        "POST",
        f"{console_api}/login",
        json={"email": email, "password": pwd_b64},
    )
    data = extract_data(payload)
    token = data.get("access_token") if isinstance(data, dict) else None
    if not token:
        raise RuntimeError(f"Login failed, response={payload}")
    return token

def safe_filename(name: str) -> str:
    name = re.sub(r"[\\/:*?\"<>|]+", "_", name).strip()
    return name[:120] or "app"

def list_all_apps(console_api: str, token: str) -> List[Dict[str, Any]]:
    apps: List[Dict[str, Any]] = []
    page, limit = 1, 30
    while True:
        payload = http_json("GET", f"{console_api}/apps", token, params={"page": page, "limit": limit})
        data = extract_data(payload)
        batch = data.get("data", []) if isinstance(data, dict) else []
        total = data.get("total", 0) if isinstance(data, dict) else 0
        apps.extend(batch)
        if len(apps) >= total:
            break
        page += 1
    return apps

def export_app_dsl(console_api: str, token: str, app_id: str, include_secret: bool = False) -> str:
    payload = http_json(
        "GET",
        f"{console_api}/apps/{app_id}/export",
        token,
        params={"include_secret": "true" if include_secret else "false"},
    )
    # Medium示例里 data 字段就是 yml 字符串citeturn1view2
    yml_text = extract_data(payload)
    if not isinstance(yml_text, str) or not yml_text.strip():
        raise RuntimeError(f"Export DSL failed: app_id={app_id}, resp={payload}")
    return yml_text

def get_model_providers(console_api: str, token: str) -> Any:
    payload = http_json("GET", f"{console_api}/workspaces/current/model-providers", token)
    return extract_data(payload)

def get_models_by_type(console_api: str, token: str, model_type: str) -> Any:
    payload = http_json("GET", f"{console_api}/workspaces/current/models/model-types/{model_type}", token)
    return extract_data(payload)

def get_plugins(console_api: str, token: str) -> Any:
    # 如果你的实例未启用插件系统，这个接口可能 404/报错；上层捕获即可
    payload = http_json("GET", f"{console_api}/workspaces/current/plugin/list", token)
    return extract_data(payload)

def try_get_version(console_api: str, token: str) -> Any:
    # /version 在不同部署下可能需要 current_version 参数citeturn15view0
    try:
        payload = http_json("GET", f"{console_api}/version", token, params={"current_version": "0.0.0"})
        return extract_data(payload)
    except Exception:
        return None

def analyze_dsl(yml_text: str) -> Dict[str, Any]:
    try:
        obj = yaml.safe_load(yml_text)
    except Exception:
        return {"parse_ok": False, "node_types": [], "llm_models": []}

    nodes = (((obj or {}).get("workflow") or {}).get("graph") or {}).get("nodes") or []
    node_types = []
    llm_models: List[Tuple[str, str]] = []
    if isinstance(nodes, list):
        for n in nodes:
            if not isinstance(n, dict):
                continue
            t = n.get("type")
            if t:
                node_types.append(str(t))
            if t == "llm":
                model = (((n.get("data") or {}).get("model")) or {})
                if isinstance(model, dict):
                    prov = str(model.get("provider") or "")
                    name = str(model.get("name") or "")
                    if prov or name:
                        llm_models.append((prov, name))
    return {
        "parse_ok": True,
        "node_types": sorted(set(node_types)),
        "llm_models": sorted(set(llm_models)),
        "nodes_count": len(nodes) if isinstance(nodes, list) else 0,
    }

def main():
    dify_origin = os.environ["DIFY_ORIGIN"].rstrip("/")
    # 如果你单独配置了 CONSOLE_API_URL（不同域名），可用 DIFY_CONSOLE_API 覆盖
    console_api = os.environ.get("DIFY_CONSOLE_API", f"{dify_origin}/console/api").rstrip("/")
    email = os.environ["DIFY_EMAIL"]
    password = os.environ["DIFY_PASSWORD"]

    out_dir = pathlib.Path("out")
    dsl_dir = out_dir / "dsl_exports"
    out_dir.mkdir(parents=True, exist_ok=True)
    dsl_dir.mkdir(parents=True, exist_ok=True)

    token = login(console_api, email, password)

    version_info = try_get_version(console_api, token)

    providers = get_model_providers(console_api, token)
    llm_models = get_models_by_type(console_api, token, "llm")
    rerank_models = get_models_by_type(console_api, token, "rerank")
    embed_models = get_models_by_type(console_api, token, "text-embedding")

    try:
        plugins = get_plugins(console_api, token)
    except Exception:
        plugins = {"error": "plugins endpoint unavailable (plugin system disabled or API changed)"}

    apps = list_all_apps(console_api, token)

    coverage_rows = []
    best = None

    for app in apps:
        app_id = app.get("id")
        app_name = app.get("name") or app_id
        if not app_id:
            continue
        yml_text = export_app_dsl(console_api, token, app_id, include_secret=False)
        fname = safe_filename(app_name) + f"__{app_id}.yml"
        (dsl_dir / fname).write_text(yml_text, encoding="utf-8")

        info = analyze_dsl(yml_text)
        score = len(info.get("node_types", [])) + 0.2 * len(info.get("llm_models", []))
        row = {
            "app_id": app_id,
            "app_name": app_name,
            "score": score,
            "node_types": info.get("node_types", []),
            "llm_models": info.get("llm_models", []),
            "nodes_count": info.get("nodes_count", 0),
            "parse_ok": info.get("parse_ok", False),
            "file": str((dsl_dir / fname).as_posix()),
        }
        coverage_rows.append(row)
        if best is None or row["score"] > best["score"]:
            best = row

    coverage_rows.sort(key=lambda x: x["score"], reverse=True)

### 10. 方案详述（来源：Dify 定时任务工作流）

## 1) Dify v1.12.1 是否“原生支持定时任务触发工作流”？

就 **开源版 Dify v1.12.1** 的常见能力来说：**一般不提供“内置 Cron/定时触发器”去自动运行某个 Workflow App**（工作流通常由 Chat/API 触发）。  
因此要实现“每天自动跑一次”，最稳的做法是：

- **用外部调度器**（Linux `cron` / `systemd timer` / Jenkins / GitHub Actions 等）  
- **每天定时调用 Dify 的 Workflow 运行 API**（相当于“定时触发工作流”）

> 你也可以在 Dify 的工作流编辑器里看看是否存在类似 **Trigger / Schedule / Cron** 的配置项：  
> - **如果 UI 里就有 Schedule/Cron**：那说明你当前版本/分支/插件已经支持原生定时触发；  
> - **如果没有**：就按外部 cron 调用 API 的方式做（下面我给你完整方案）。

---

## 2) 交付：每天抓取全球 AI 重要新闻并汇总成报告的 Workflow App DSL（可导入）

> 说明（很重要）：你要“开箱即用”且“不依赖额外 Key”，最稳的抓取源是 **Google News RSS**（无需 API Key）。  
> 该 DSL 工作流会：
> 1) 拉取多组 AI 关键词 RSS  
> 2) 去重、按“出现次数 + 新鲜度”排序  
> 3) 让 LLM 生成日报（Markdown）

把下面内容保存为 `daily_ai_news_report_dify.dsl.yml`，在 Dify 里 **Import DSL** 导入即可。

```yaml
version: 0.1.0
kind: app
app:
  name: "Daily AI News Report"
  mode: workflow
  icon: "🗞️"
  icon_background: "#EEF2FF"
  description: "每天抓取全球AI要闻（Google News RSS）并汇总为Markdown日报。建议用外部cron定时调用本工作流API运行。"
  use_icon_as_answer_icon: true

workflow:
  features:
    file_upload:
      enabled: false

      enabled: false

    nodes:
      - id: start
        type: start

              label: "Report Date (optional)"
              type: string
              required: false
              default: ""
            - variable: language
              label: "Language"
              type: select
              required: true
              options:
                - label: "中文"
                  value: "zh"
                - label: "English"
                  value: "en"
              default: "zh"
            - variable: max_items
              label: "Max Items"
              type: number
              required: true
              default: 20

      - id: fetch_rss
        type: code
        position:
          x: 360
          y: 120
        data:
          title: "Fetch Google News RSS"
          language: python3
          code: |
            import re
            import json
            import time
            import xml.etree.ElementTree as ET
            from urllib.parse import quote_plus, urlparse, parse_qsl, urlencode, urlunparse
            from urllib.request import Request, urlopen
            from email.utils import parsedate_to_datetime
            from datetime import datetime, timezone

            def _clean_url(u: str) -> str:
              # 去掉常见跟踪参数，提升去重效果
              try:
                p = urlparse(u)
                q = [(k, v) for (k, v) in parse_qsl(p.query, keep_blank_values=True)
                     if k.lower() not in {"utm_source","utm_medium","utm_campaign","utm_term","utm_content","gclid","fbclid"}]
                return urlunparse((p.scheme, p.netloc, p.path, p.params, urlencode(q), p.fragment))
              except Exception:
                return u

            def _fetch(url: str, timeout=20) -> str:
              req = Request(url, headers={"User-Agent":"Mozilla/5.0"})
              with urlopen(req, timeout=timeout) as r:
                return r.read().decode("utf-8", errors="ignore")

            def _rss_url(query: str, hl="en-US", gl="US", ceid="US:en"):
              return f"https://news.google.com/rss/search?q={quote_plus(query)}&hl={hl}&gl={gl}&ceid={ceid}"

            def main(inputs: dict) -> dict:
              language = inputs.get("language", "zh")
              max_items = int(inputs.get("max_items", 20) or 20)

              # 关键词可自行增删：覆盖“产品发布/融资并购/政策监管/研究突破/安全事件”
              queries = [

                "OpenAI",

                "AI data breach",
                "AI startup funding",
                "AI model release"

              rss_urls = [_rss_url(q) for q in queries]

              items = []
              now = datetime.now(timezone.utc)

              for q, url in zip(queries, rss_urls):
                try:
                  xml_text = _fetch(url, timeout=25)
                  root = ET.fromstring(xml_text)
                  for it in root.findall(".//item"):
                    title = (it.findtext("title") or "").strip()
                    link = (it.findtext("link") or "").strip()
                    pub = (it.findtext("pubDate") or "").strip()
                    source_el = it.find("source")
                    source = (source_el.text or "").strip() if source_el is not None else ""
                    desc = (it.findtext("description") or "").strip()
                    snippet = re.sub(r"<[^>]+>", "", desc).strip()

                    if not title or not link:
                      continue

                    published_ts = None
                    try:
                      dt = parsedate_to_datetime(pub)
                      if dt.tzinfo is None:
                        dt = dt.replace(tzinfo=timezone.utc)
                      published_ts = int(dt.timestamp())
                    except Exception:
                      published_ts = None

                    items.append({
                      "title": title,
                      "url": _clean_url(link),
                      "published": pub,
                      "published_ts": published_ts,
                      "source": source,
                      "snippet": snippet[:500],
                      "hit_query": q
                    })
                  time.sleep(0.2)
                except Exception as e:
                  items.append({
                    "error": str(e),
                    "url": url,
                    "hit_query": q
                  })

              # 去重 + 排序：同一URL命中多个query更重要；越新越靠前
              by_url = {}
              for it in items:
                if "url" not in it or not it.get("url") or it.get("error"):
                  continue
                u = it["url"]
                if u not in by_url:
                  by_url[u] = {
                    "title": it.get("title",""),
                    "url": u,
                    "source": it.get("source",""),
                    "published": it.get("published",""),
                    "published_ts": it.get("published_ts"),
                    "snippets": [],
                    "hit_queries": set()
                  }
                by_url[u]["snippets"].append(it.get("snippet",""))
                by_url[u]["hit_queries"].add(it.get("hit_query",""))

                # 保留最新发布时间
                if it.get("published_ts") and (by_url[u].get("published_ts") is None or it["published_ts"] > by_url[u]["published_ts"]):
                  by_url[u]["published_ts"] = it["published_ts"]
                  by_url[u]["published"] = it.get("published","")

              merged = []
              for u, v in by_url.items():
                merged.append({
                  "title": v["title"],
                  "url": v["url"],
                  "source": v["source"],
                  "published": v["published"],
                  "published_ts": v["published_ts"],
                  "hit_count": len(v["hit_queries"]),
                  "hit_queries": sorted(list(v["hit_queries"]))[:6],
                  "snippet": " / ".join([s for s in v["snippets"] if s])[:800]
                })

              def score(x):
                hit = x.get("hit_count", 1)
                ts = x.get("published_ts") or 0
                # 新鲜度：按“距今小时”做衰减（没有时间就当旧）
                age_hours = (now.timestamp() - ts) / 3600.0 if ts else 9999
                freshness = max(0.0, 1.0 - (age_hours / 72.0))  # 72小时内更优
                return hit * 10.0 + freshness
